function copyCode(button) {
  const code = button.nextElementSibling.innerText.trim();
  navigator.clipboard.writeText(code).then(() => {
    button.innerText = 'Copied!';
    setTimeout(() => { button.innerText = 'Copy'; }, 1500);
  });
}
